# 重定向、别名

## 重定向 redirect

三种书写形式：

```ts
const routes: Array<RouteRecordRaw> = [
  {
    path: '/',
    component: () => import('../components/root.vue'),
    // redirect: '/a', // 1. 字符串形式
    // redirect:{
    //   path:'/a' // 2. 对象形式，可以用path，也可以用name
    // }
    redirect: to=>{ // 3. 回调函数形式，可以传参
      // return '/a'  // 直接返回字符串路径
      return {
        path: '/a' ,
        query: to.query
      }
    }
    children: [
      {
        path: '/a',
        component: () => import('../components/A.vue'),
      },
    ],
  },
]
```

## 别名 alias

```ts
const routes: Array<RouteRecordRaw> = [
  {
    path: '/',
    component: () => import('../components/root.vue'),
    alias: ['/root1', '/root2'],
  },
]
```
