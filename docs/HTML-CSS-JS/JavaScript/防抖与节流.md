# 防抖与节流

## 手写实现

```ts
const debounce = (func: Function, delay = 200) => {
  let timer: NodeJS.Timeout
  return (...rest: any[]) => {
    timer && clearTimeout(timer)
    timer = setTimeout(() => {
      func(...rest)
    }, delay)
  }
}

const throttle = (func: Function, delay = 200) => {
  let timer: NodeJS.Timeout | null
  return (...rest: any[]) => {
    if (!timer) {
      timer = setTimeout(() => {
        func(...rest)
        timer = null
      }, delay)
    }
  }
}
```
